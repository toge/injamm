- [ ] **Step 1: クラス名を `engine` に変更**

```cpp
template <class T>
class engine {
  detail::bytecode bc_;

public:
  engine() = default;

  explicit engine(std::string_view tmpl) : bc_(detail::bc_compile<T>(tmpl)) {}

  [[nodiscard]] expected<std::string> render(T const& value) const {
    return detail::bc_execute(bc_, value);
  }
};
```
